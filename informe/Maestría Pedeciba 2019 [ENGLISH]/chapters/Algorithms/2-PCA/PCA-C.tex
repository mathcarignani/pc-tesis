\beginAlgorithm
\inputAndOutput{\cInputFile, \cInputWindow, \cInputThreshold}{\cOutputFile{\coderPCA}}
\createBinary\\
\codeAlgoCode{\intPCA}\\
\codeWindow\\
\codeHeader\\
\codeDataRows\\
\ForEach{\forEachColumnCoder}{
    \setError\\
    \createWindow\\
    \ForEach{\forEachEntryCoder}{
        \windowPush{\entry}\\
        \If{\windowNotFull}{
            \continueAlgo
        }
        \uIf{\windowNoData}{
            \codeBit{0}\\
            \codeNoData
        }
        \uElseIf{\windowInt \andd \validThre}{
            $\valuev = \windowMin + \windowMax$\\
            \If{$\valuev \neq 0$}{
                $\valuev = \valuev /2$\\
            }
            \codeBit{0}\\
            \codeValuePCA
        }
        \Else{
            \codeBit{1}\\
            \ForEach{\windowVal \textnormal{ in} $\window.\textnormal{values}$}{
                \eIf{\ifNoDataCoder{\windowVal}}{
                    $\valuev = \nodatainteger$
                }
                {
                    $\valuev = \plusOffset{\windowVal}$
                }
                \codeValueBase\\
            }
        }
        \createWindow\\
    }
}
\closeFile
\EndAlgorithm{\coderPCA}{\label{pseudoCoderPCA}}
