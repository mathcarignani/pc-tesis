\setlength{\algomargin}{1em}
\IncMargin{1em}
\begin{figure}[h]
\begin{algorithm}[H]
\SetAlgoLined
  \SetKwInOut{Input}{input}
  \SetKwInOut{Output}{output}
\Indm  
  \SetAlgoLined
  \Input{
            $\file \in \textit{Datasets}$: csv file to be coded\\
        }
  \Output{\coderBase(\file): binary file coded with \coderBase}
\Indp
    \createbinary\\
    \out.\codeint\text{(\textit{CODER\_BASE}, 8)}\\
    \out.\text{code\_header(\file)}\\
    \out.\codeint\text{(\file.\datarowscount(), 24)}\\
    \ForEach{\column \textnormal{ in \file.columns}}{
        \ForEach{\entry \textnormal{ in} $\column.\textnormal{entries}$}{
            \eIf{$\entry = \textit{NO\_DATA}$\xspace}{
                $\valuev = \nodatainteger$
            }
            {
                $\valuev = \entry + \column.\text{offset}$
            }
            \out.\text{code\_integer}(\valuev, \column.\text{total\_bits})\\
        }
    }
    \closefile
\end{algorithm}
\vspace{-15pt}
\caption{\coderBase \ pseudocode.}
\label{pseudoCoderBase}
\end{figure}

\IncMargin{1em}
\begin{figure}[h]
\begin{algorithm}[H]
\SetAlgoLined
  \SetKwInOut{Input}{input}
  \SetKwInOut{Output}{output}
\Indm  
  \SetAlgoLined
  \Input{\file: binary file coded with \coderBase}
  \Output{\decoderBase(\file): csv file decoded with \decoderBase}
\Indp
    \createcsv\\
    $\textit{coder\_value} = \file.\decodeint(8)$\\
    \out.\text{decode\_header(\file)}\\
    $\datarowscount = \file.\decodeint(24)$\\
    \eIf{\textit{coder\_value} = \textit{CODER\_BASE}}{
        \ForEach{\column \textnormal{ in \out.columns}}{
            \ForEach{\entry \textnormal{ in} $\column.\textnormal{entries}$}{
                $\valuev = \file.\text{decode\_integer}(\column.\text{total\_bits})$\\
                \eIf{$\valuev = \nodatainteger$\xspace}{
                    \out.\text{write\_string}(\textit{NO\_DATA})\\
                }
                {
                    \out.\text{write\_string}($\valuev - \column.\text{offset}$)\\
                }
            }
        }
    }
    {
    . . . \text{// if \file \ was coded with a different coder}
    }
    \closefile
\end{algorithm}
\vspace{-15pt}
\caption{\decoderBase \ pseudocode.}
\label{pseudoDecoderBase}
\end{figure}
